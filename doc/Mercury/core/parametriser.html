164	def apply_alpha_tat_mean(self, alpha):
165		"""
166		We modify the mean so as sig/mean is constant.
167		Mean of expo((x-loc)/lambda) is lambda + loc. Sdt is lambda.
168		"""
169
170		for airport in self.world.airports.values():
171			new_dists = {}
172			for k in airport.tats_old.keys():
173				new_dists[k] = {}
174				for kk, vv in airport.tats_old[k].items():
175					loc, l = vv.kwds['loc'], vv.kwds['scale']
176					loc, l = alpha * loc, alpha * l
177					new_dists[k][kk] = expon(loc=loc, scale=l)
178
179			airport.give_turnaround_time_dists(new_dists)
180
181	def apply_alpha_mct(self, alpha):
182		"""
183		For now, pushes the distribution via loc.
184		"""
185
186		for airport in self.world.airports.values():
187			new_dists = {}
188			for pax_type in airport.mcts_old.keys():
189				new_dists[pax_type] = {}
190				for connection, old_dist in airport.mcts_old[pax_type].items():
191					loc, scale, s = old_dist.kwds['loc'], old_dist.kwds['scale'], old_dist.kwds['s']
192					mu = old_dist.mean()
193					mu_p = alpha * mu
194					loc += mu_p - mu
195					
196					new_dists[pax_type][connection] = lognorm(loc=loc, scale=scale, s=s)
197
198			airport.give_connecting_time_dist(new_dists)
199
200	def apply_delta_mct(self, delta):
201		"""
202		For now, pushes the distribution via loc.
203		"""
204
205		for airport in self.world.airports.values():

165	def apply_alpha_tat_mean(self, alpha):
166		"""
167		We modify the mean so as sig/mean is constant.
168		Mean of expo((x-loc)/lambda) is lambda + loc. Sdt is lambda.
169		"""
170
171		for airport in self.world.airports.values():
172			new_dists = {}
173			for k in airport.tats_old.keys():
174				new_dists[k] = {}
175				for kk, vv in airport.tats_old[k].items():
176					loc, l = vv.kwds['loc'], vv.kwds['scale']
177					loc, l = alpha * loc, alpha * l
178					new_dists[k][kk] = expon(loc=loc, scale=l)
179
180			airport.give_turnaround_time_dists(new_dists)

183		"""
184		For now, pushes the distribution via loc.
185		"""
186
187		for airport in self.world.airports.values():
188			new_dists = {}
189			for pax_type in airport.mcts_old.keys():

192					loc, scale, s = old_dist.kwds['loc'], old_dist.kwds['scale'], old_dist.kwds['s']
193					mu = old_dist.mean()
194					mu_p = alpha * mu

196					
197					new_dists[pax_type][connection] = lognorm(loc=loc, scale=scale, s=s)

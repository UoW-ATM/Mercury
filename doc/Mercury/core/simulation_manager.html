167	def post_process_paras(self, paras):
168		if paras['network_manager__ATFM_regulation_mechanism'] is not None:
169			if paras['network_manager__ATFM_regulation_mechanism'] == 'UDPP':
170				paras['network_manager__hotspot_solver'] = {'global': 'udpp_merge', 'local': 'udpp_local'}
171			elif paras['network_manager__ATFM_regulation_mechanism'] == 'UDPP+ISTOP':
172				paras['network_manager__hotspot_solver'] = {'global': 'udpp_istop', 'local': 'function_approx'}
173				paras['network_manager__hotpost_archetype_function'] = 'jump2'
174			elif paras['network_manager__ATFM_regulation_mechanism'] == 'CM':
175				paras['network_manager__modules'].insert(0, 'CM')  # ['HMI_FP_SEL'] # 'FAC_FIFO_queue']#['FAC_total_cost']#['FAC_total_delay']##
176				# paras['hotspot_solver'] = {'global':'nnbound', 'local':'function_approx'}
177				paras['network_manager__hotspot_solver'] = {'global': 'globaloptimum', 'local': 'function_approx'}
178				paras['network_manager__hotpost_archetype_function'] = 'jump2'
179			elif paras['network_manager__ATFM_regulation_mechanism'] == 'UDPP+ISTOP_TRUE':
180				paras['network_manager__hotspot_solver'] = {'global': 'udpp_istop', 'local': 'get_cost_vectors'}
181				paras['network_manager__hotpost_archetype_function'] = None
182			elif paras['network_manager__ATFM_regulation_mechanism'] == 'GLOBAL_TRUE':
183				paras['network_manager__hotspot_solver'] = {'global': 'globaloptimum', 'local': 'get_cost_vectors'}
184				paras['network_manager__hotpost_archetype_function'] = None
185			elif paras['network_manager__ATFM_regulation_mechanism'] == 'GLOBAL_APPROX':
186				paras['network_manager__hotspot_solver'] = {'global': 'globaloptimum', 'local': 'function_approx'}
187				paras['network_manager__hotpost_archetype_function'] = 'jump2'
188			elif paras['network_manager__ATFM_regulation_mechanism'] == 'NNBOUND_TRUE':
189				paras['network_manager__hotspot_solver'] = {'global': 'nnbound', 'local': 'get_cost_vectors'}
190				paras['network_manager__hotpost_archetype_function'] = None
191			elif paras['network_manager__ATFM_regulation_mechanism'] == 'NNBOUND_APPROX':

193				paras['network_manager__hotpost_archetype_function'] = 'jump2'
194			else:
195				raise Exception('Unknown mechanism {}'.format(paras['network_manager__ATFM_regulation_mechanism']))

199	def _run_one_iteration(self, info_scenario={}, case_study_conf={}, data_scenario={}, paras_scenario={},
200						   paras_simulation={},
201						   n_iter=None, connection_write={}, connection_read={}, world=None, parametriser=None,
202						   results_aggregator=None,
203						   ):
204		"""
205		Runs one iteration of mercury with the world and parameters passed. Can pass iterations if output folder
206		exists. Can dump all results or just compute a final dataframe with the results aggregator.
207
208		The rebuild_agents is here to control whether agents should be rebuilt. In general, the world is not destroyed
209		at the end of an iteration but undergoes a "deep cleaning", with all agents destroyed. These agents need to be
210		built again before the next iteration. Note that if one wants persistance of agents across several iterations,
211		this is not supported at the moment (see the reset method in the world builder). Note also that world do not need
212		to be built every iteration. Keeping the world allows to save loading time. Note that random variables involved
213		in the scenario generation (like days of regulation etc) are redrawn by default when the world is reset.
214		"""
215		# TODO: put this in the world builder
216		paras_scenario = self.post_process_paras(paras_scenario)
217
218		# If the results are supposed to go in files (not in mysql), we check
219		# if the files already exist and we are instructed to skip them if so.
220		# If the simulation is skipped, the results are loaded in the world object,
221		# allowing for instance the results aggregator to get all results at once.
222		save_path_relative = self.compute_path(paras_simulation=paras_simulation,
223											   info_scenario=info_scenario,
224											   info_case_study=case_study_conf['info'],
225											   paras_scenario=paras_scenario,
226											   n_iter=n_iter)
227
228		if save_path_relative is None:
229			full_path = None
230		else:
231			if connection_write is not None:
232				full_path = connection_write['base_path'] / save_path_relative
233			else:
234				full_path = save_path_relative
235
236		if (full_path is not None and not full_path.exists()) or (
237				not paras_simulation['outputs_handling__skip_computation_if_results_exists']):
238
239			if world is None:
240				print('WORLD IS BUILT IN RUN ONE ITERATION (1)')
241				world = self.build_world(connection=connection_read,
242										 parametriser=parametriser,
243										 case_study_conf=case_study_conf,
244										 info_scenario=info_scenario,
245										 data_scenario=data_scenario,
246										 paras_scenario=paras_scenario,
247										 paras_simulation=paras_simulation)
248
249			if not world.agents_built:
250				self.build_agents(parametriser=parametriser,
251								  world=world,
252								  )
253
254			if parametriser is not None:
255				parametriser.apply_all_values_post_load(paras_scenario)
256
257			world.run_world(n_iter)
258			if not paras_simulation['outputs_handling__skip_results']:
259				world.get_all_metrics()
260				world.dump_all_results(n_iter,
261									   connection_write,
262									   paras_simulation['write_profile'],
263									   full_path)
264
265		else:
266			print('Skipping computation because files already exist (loading from disk instead)...\n')
267
268			# Load data from disk.
269			for output in paras_simulation['outputs']:
270				stuff = output.split('output_')[-1]
271
272				file_name = output + str('.csv.gz')
273
274				setattr(world,
275						'df_' + stuff,
276						read_data(path=full_path,
277								  file_name=file_name,
278								  connection=connection_write
279								  )
280						)
281
282		if results_aggregator is not None:
283			results = results_aggregator.compute_results_individual_iteration(info_scenario['scenario_id'], case_study_conf['info']['case_study_id'], n_iter, world, paras_scenario)
284		else:
285			results = None
286
287		return results

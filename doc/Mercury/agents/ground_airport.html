130	def give_connecting_time_dist(self, dists, mct_q=0.95):
131		"""
132		Initialise connecting time distribution for the agent
133		dists is a single layer dict with pax_type keys.
134		Connecting times vary as a function of type of connection
135		(National-National), (International-International), Other
136		"""
137		self.connecting_time_dists = dists
138
139		# Get minimum connecting times
140		self.mcts = {}
141		self.connecting_time_dists = {}
142		for pax_type, v in dists.items():
143			self.mcts[pax_type] = {}
144			self.connecting_time_dists[pax_type] = {}
145			for connection, dist in v.items():
146				if (connection == 'N-N') or (connection == (False, False)):
147					connection_type = (False, False)
148					self.mcts[pax_type][connection_type] = dist.ppf(mct_q)
149					self.connecting_time_dists[pax_type][connection_type] = dist
150				elif (connection == 'I-I') or (connection == (True, True)):
151					connection_type = (True, True)
152					self.mcts[pax_type][connection_type] = dist.ppf(mct_q)
153					self.connecting_time_dists[pax_type][connection_type] = dist
154				else:
155					connection_type = (True, False)
156					self.mcts[pax_type][connection_type] = dist.ppf(mct_q)
157					self.connecting_time_dists[pax_type][connection_type] = dist
158
159					connection_type = (False, True)
160					self.mcts[pax_type][connection_type] = dist.ppf(mct_q)
161					self.connecting_time_dists[pax_type][connection_type] = dist

216	"""
217	GH: Ground Handler
218
219	Description:
220		- Provides the turnaround time for an aircraft and
221		- Does the turnaround process by blocking the aircraft resource while the
222		turnaround is happening.
223	"""

225	def compute_turnaround_time(self, ac_wake, ao_type):
226		"""
227		Provides turnaround time as a function of aircraft category (wake turbulence) and airline type (FSC, LCC, etc.)
228		"""
229		turnaround_time = self.agent.turnaround_time_dists[ac_wake][ao_type].rvs(random_state=self.agent.rs)
230		return turnaround_time
231
232	def do_turnaround(self, aircraft, tt):
233		"""
234		Block the aircraft resource is blocked while the turnaround process it taking place at the airport
235		"""
236		mprint(aircraft, 'waits at', self.agent, 'for', tt, 'minutes starting at t=', self.agent.env.now)
237		# Wait turnaround is finished
238		yield self.agent.env.timeout(tt)
239
240		# Release resource for next flight
241		aircraft.release(aircraft.users[0])
242		mprint(self.agent, 'releases', aircraft, 'at t=', self.agent.env.now)
243
244	def send_turnaround_time(self, aoc_uid, flight_uid, tt):
245		mprint(self.agent, 'sends turnaround time for flight', flight_uid, ':', tt)
246		if flight_uid in flight_uid_DEBUG:
247			print("{} send turnaround time for flight {}".format(self.agent, flight_uid))
248		msg_back = Letter()
249		msg_back['to'] = aoc_uid
250		msg_back['type'] = 'turnaround_time'
251		msg_back['body'] = {'flight_uid': flight_uid,
252							'turnaround_time': tt}
253		self.send(msg_back)        
254	
255	def wait_for_turnaround_request(self, msg):
256		"""
257		Do the turnaround for a given aircraft
258		"""
259		mprint(self.agent, 'received turnaround request from AOC', msg['from'])
260
261		if msg['body']['flight_uid'] in flight_uid_DEBUG:
262			print("{} receives turnaround_time request for flight {}".format(self.agent, msg['body']['flight_uid']))
263
264		aircraft = msg['body']['aircraft']
265
266		mprint('Flight', msg['body']['flight_uid'], 'waits for turnaround of', aircraft)
267		tt = self.compute_turnaround_time(aircraft.bada_performances.wtc, msg['body']['ao_type'])
268		
269		self.agent.env.process(self.do_turnaround(aircraft, tt))
270
271		# This is normal, message gets transferred by flight to AOC afterwards.
272		aoc_uid = aircraft.get_next_flight()
273
274		self.send_turnaround_time(aoc_uid,
275									msg['body']['flight_uid'],
276									tt)
277
278	def wait_for_turnaround_time_request(self, msg):
279		mprint(self.agent, 'received turnaround time request from AOC', msg['from'])
280
281		tt = self.compute_turnaround_time(msg['body']['ac_type'], msg['body']['ao_type'])
282		self.send_turnaround_time(msg['from'], msg['body']['flight_uid'], tt)

367	def compute_taxi_out_time(self, ac_type, ao_type, taxi_out_time_estimation):
368		"""
369		Sample the taxi-out time from the distribution
370		"""
371		# Note that taxi-out does not depend on aircraft type (ac_type) nor airline operator type (ao_type) for now
372		taxi_out_time = taxi_out_time_estimation + self.agent.taxi_time_add_dists.rvs(random_state=self.agent.rs)
373		return max(self.agent.min_tt, taxi_out_time)

396	def compute_taxi_in_time(self, ac_type, ao_type):
397		"""
398		Note: we add the "estimation" and the "disruption" in order to have
399		the same distribution as for taxi-out.
400		"""
401		# Note that taxi-in does not depend on ac_type or ao_type for now
402		estimation = self.agent.taxi_in_time_estimation_dists.rvs(random_state=self.agent.rs)
403		disruption = self.agent.taxi_time_add_dists.rvs(random_state=self.agent.rs)
404		return max(self.agent.min_tt, estimation + disruption)

409		msg_back['type'] = 'taxi_in_time'
410		msg_back['body'] = {'flight_uid': flight_uid,
411							'taxi_in_time': ti
412							}

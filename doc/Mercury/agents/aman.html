318	def create_slot_model_pyomo(costs, flight_min_max_constraints={}):
319		model = pyo.ConcreteModel()
320		model.SLOTS = pyo.RangeSet(0, costs.shape[1]-1)
321		model.FLIGHTS = pyo.RangeSet(0, costs.shape[0]-1)
322		model.COSTS = costs
323
324		model.y = pyo.Var(model.SLOTS, model.FLIGHTS, within=pyo.Binary)
325
326		# MinMaxArrivalTimes
327		for f, v in flight_min_max_constraints.items():
328			for s in range(0, v[0]):
329				model.y[s, f].fix(0)
330			for s in range(v[1]+1, costs.shape[1]):
331				model.y[s, f].fix(0)
332
333		def _ObjFunc(model):
334			return sum(model.y[s, f]*model.COSTS[f, s] for (s, f) in model.y)
335
336		model.obj = pyo.Objective(rule=_ObjFunc, sense = pyo.minimize)
337
338		def _FlightCon(model, f):
339			return sum(model.y[s, f] for s in model.SLOTS) == 1
340		model.FlightCon = pyo.Constraint(model.FLIGHTS, rule=_FlightCon)
341
342		def _SlotCon(model, s):
343			return sum(model.y[s, f] for f in model.FLIGHTS) <= 1
344		model.SlotCon = pyo.Constraint(model.SLOTS, rule=_SlotCon)
345
346		return model

372	def create_model_google(costs, f_min_max_cnt):
373		solver = pywraplp.Solver('slot_assigment',
374								 pywraplp.Solver.CBC_MIXED_INTEGER_PROGRAMMING)
375
376		var = np.empty(costs.shape, dtype=pywraplp.Variable)
377		objective = solver.Objective()
378		f_constraints = [solver.Constraint(1, 1) for f in range(var.shape[0])]
379		s_constraints = [solver.Constraint(0, 1) for s in range(var.shape[1])]
380
381		for s in range(costs.shape[1]):
382			for f in range(costs.shape[0]):
383				if (s < f_min_max_cnt[f][0]) or (s > f_min_max_cnt[f][1]):
384					v_max = 0
385					v_min = 0
386				else:
387					v_max = 1
388					v_min = 0
389
390				# if f==0:
391				#    print(f,s,(v_min,v_max))
392
393				var[f, s] = solver.IntVar(v_min, v_max, str(f)+"_"+str(s))
394
395				objective.SetCoefficient(var[f][s], float(costs[f][s].item()))
396
397				f_constraints[f].SetCoefficient(var[f][s], 1)
398				s_constraints[s].SetCoefficient(var[f][s], 1)
399
400		return solver, var

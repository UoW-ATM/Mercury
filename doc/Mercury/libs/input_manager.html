 15def read_data_from_dict(dictionary,key,data_path):
 16
 17    new_dict = deepcopy(dictionary[key])
 18
 19    def traverse(d, ppath=Path()):
 20        for k, v in d.items():
 21            if isinstance(v, dict):
 22                traverse(v, ppath / k)
 23            else:
 24                #print('reading: ',str(ppath / str(v+'.parquet')))
 25                try:
 26                    d[k] = pd.read_parquet(ppath / str(v+'.parquet'))
 27                except FileNotFoundError:
 28                    #print(v, 'does not exist')
 29                    d[k] = v
 30        return d
 31
 32
 33    return traverse(new_dict,data_path / key)
